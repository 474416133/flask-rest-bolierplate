# flask rest bolierplate


```sh
# 虚拟环境
python3 -m venv venv

# 激活
. venv/bin/activate

# 安装依赖
pip install Flask
pip install Flask-SQLAlchemy
pip install flask-marshmallow
pip install marshmallow-sqlalchemy

# 环境变量
export FLASK_APP=flaskr
export FLASK_ENV=development

# 初始化
flask init-db

# 运行
flask run

# 打开一个 Shell
flask shell
```
